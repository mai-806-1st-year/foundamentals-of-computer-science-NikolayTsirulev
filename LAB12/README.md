### 8. Распечатка протокола
```
#include <stdio.h>
#include <stdbool.h>
#include <ctype.h>

bool isEven(char k) {
    if ((k - '0') % 2 == 0) return true;
    else return false;
}
typedef enum {
    OUT,
    IN_DIGIT
} state;


int main() {
    state state = OUT;
    for (char a = getchar(); a != EOF; a = getchar()) {
        switch (state) {
            case OUT:
                if (isdigit(a)) state = IN_DIGIT;
                else {
                    putchar(a);
                    state = OUT;
                    break;
                }
            case IN_DIGIT:
                if (isEven(a)) {
                    printf("%d", a - '0' + 1);
                } else putchar(a);
                state = OUT;
                break;
        }
    }
    return 0;
}
```
